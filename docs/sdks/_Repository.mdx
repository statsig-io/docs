import DarkLogo from "@site/static/img/GitHub-Mark-64px.png";
import LightLogo from "@site/static/img/GitHub-Mark-Light-64px.png";
import { useColorMode } from "@docusaurus/theme-common";

export default function ({ repo }) {
  const isDarkTheme = useColorMode().colorMode === "dark";
  return (
    <div
      style={{
        marginTop: -20,
        display: "flex",
      }}
    >
      <a
        href={repo}
        style={{
          fontSize: 14,
          fontWeight: "600",
          color: isDarkTheme ? "white" : "black",
          display: "flex",
          alignItems: "center",
        }}
      >
        <img
          src={isDarkTheme ? LightLogo : DarkLogo}
          style={{
            height: 18,
            width: 18,
            marginRight: 6,
          }}
        />
        {repo.replace("https://github.com/", "")}
      </a>
    </div>
  );
}

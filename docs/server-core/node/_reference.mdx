### Experiment Methods

```typescript
// Get experiment evaluation for a user
getExperiment(user: StatsigUser, experimentName: string): Experiment

// Get experiment configured for a specific group name
getExperimentByGroupName(user: StatsigUser, experimentName: string, groupName: string): Experiment
```

### Fields Needed Methods (Enterprise Only)

```typescript
// Get user fields needed for a gate evaluation
getFieldsNeededForGate(gateName: string): Array<string>

// Get user fields needed for a dynamic config evaluation
getFieldsNeededForDynamicConfig(configName: string): Array<string>

// Get user fields needed for an experiment evaluation
getFieldsNeededForExperiment(experimentName: string): Array<string>

// Get user fields needed for a layer evaluation
getFieldsNeededForLayer(layerName: string): Array<string>
```

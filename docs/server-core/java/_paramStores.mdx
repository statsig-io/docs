### Retrieving Parameter Values

Parameter Store provides methods for retrieving values of different types with fallback defaults.

<Tabs
  groupId="java-core-lang-choice"
  defaultValue="java"
  values={[
    {label: 'Java', value: 'java'},
    {label: 'Kotlin', value: 'kotlin'},
  ]}>
  <TabItem value="java">

```java
// String parameters
String stringValue = parameterStore.getString("string_param", "default_value");

// Boolean parameters
boolean boolValue = parameterStore.getBoolean("bool_param", false);

// Numeric parameters
double doubleValue = parameterStore.getDouble("double_param", 0.0);
int intValue = parameterStore.getInt("int_param", 0);
long longValue = parameterStore.getLong("long_param", 0L);

// Complex parameters
Map<String, Object> defaultMap = new HashMap<>();
defaultMap.put("key", "value");
Map<String, Object> mapValue = parameterStore.getMap("map_param", defaultMap);

Object[] defaultArray = new Object[]{"item1", "item2"};
Object[] arrayValue = parameterStore.getArray("array_param", defaultArray);
```

  </TabItem>
  <TabItem value="kotlin">

```kotlin
// String parameters
val stringValue = parameterStore.getString("string_param", "default_value")

// Boolean parameters
val boolValue = parameterStore.getBoolean("bool_param", false)

// Numeric parameters
val doubleValue = parameterStore.getDouble("double_param", 0.0)
val intValue = parameterStore.getInt("int_param", 0)
val longValue = parameterStore.getLong("long_param", 0L)

// Complex parameters
val defaultMap = mutableMapOf<String, Any>().apply {
    put("key", "value")
}
val mapValue = parameterStore.getMap("map_param", defaultMap)

val defaultArray = arrayOf<Any>("item1", "item2")
val arrayValue = parameterStore.getArray("array_param", defaultArray)
```

  </TabItem>
</Tabs>

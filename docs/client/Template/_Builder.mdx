import Gap from "./_Gap.mdx";

export const Builder = (sections, toc) => {
  toc.length = 0;
  return sections.map(([Section, props], index) => {
    toc.push(...Section.toc);
    return (
      <React.Fragment key={"" + index}>
        <Section.default {...props} />
        <Gap />
      </React.Fragment>
    );
  });
};
